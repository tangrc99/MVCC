185 Iterator insert(const std::string &key) {
186
187 int level = randomLevel();
188
189
190 auto start = root_;
191
192 // 从高层到底层，找到应该插入的位置
193 for (int i = MAX_L; i > level; i--) {
194 auto prev = findPrevByKey(i, key, start);
195 start = prev;
196 }
197
198 // 检查是否有重复的值
199 std::vector<SkipListNode *> prev_nodes(level, nullptr);
200
201 for (int i = level; i > 0; i--) {
202 auto prev = findPrevByKey(i, key, start);
203 if (prev->key_ == key) {
204 return Iterator(prev);
205 }
206 prev_nodes[i - 1] = prev;
207 start = prev;
208 }
209
210 auto node = new SkipListNode(key, level);
211
212 // 在检查的基础上，重新进行搜索，然后将值更替掉
213 for (int i = level; i > 0; i--) {
214
215 bool succeed = false;
216 while (!succeed) {
217
218 auto prev = findPrevByKey(i, key, prev_nodes[i - 1]); // 重新寻找，防止有更改发生
219 auto nxt = prev->getNextNode(i);
220 node->setNextNode(nxt, nullptr, i); // 注意排序，这样修改后不会存在链表中断问题
221
222 succeed = prev->setNextNode(node, nxt, i);
223 }
224 }
225
226 size_.fetch_add(1);
227
228 return Iterator(node);
229 }

336 Iterator find(const std::string &key) {
337
338 int i = MAX_L;
339
340 SkipListNode *node = root_;
341
342 while (i > 0) {
343
344 if (node->key_ == key) {
345 return node->isDeleted() ? Iterator(nullptr) : Iterator(node);
346 }
347
348 if (node->key_ > key)
349 break;
350
351 auto nxt = node->getNextNode(i);
352 if (nxt == nullptr || nxt->key_ > key) {
353 i--;
354 } else {
355 node = nxt;
356 }
357 }
358
359 return Iterator(nullptr);
360 }

368 std::pair<Iterator, Iterator> findBetween(const std::string &min = "", const std::string &max = "") {
369 Iterator start(nullptr), end(nullptr);
370
371 SkipListNode *find_pos = root_;
372
373 if (min.empty()) {
374 start = this->begin();
375 } else {
376
377 for (int i = MAX_L; i > 0; i--) {
378 find_pos = findPrevByKey(i, min, find_pos);
379 }
380 if (find_pos->key_ != min) {
381 find_pos = find_pos->getNextNode(1);
382 }
383 start = Iterator(find_pos);
384 }
385
386 if (max.empty()) {
387 end = Iterator::end();
388 } else {
389 for (int i = find_pos->level(); i > 0; i--) {
390 find_pos = findPrevByKey(i, max, find_pos);
391 }
392 end = Iterator(find_pos);
393 }
394
395 return {std::move(start), std::move(end)};

491 void compact(){
492
493 // 需要遍历每一个层级
494 for(int level = MAX_L; level > 0;level--){
495
496 auto slow = root_;

575 SkipListNode *findPrevByKey(int level, const std::string &key, SkipListNode *start) {
576 if (start->level() < level)
577 return nullptr;
578
579 auto prev = start;
580 auto node = start->getNextNode(level);
581
582 while (node != nullptr && node->key_ <= key) {
583 prev = prev->getNextNode(level);
584 node = node->getNextNode(level);
585 }
586 return prev;
587 }
